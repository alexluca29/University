#pragma once
#include <iostream>
#include "DLLANode.h"
#define dim 50

class SetIterator;
class Set
{
private:
	DLLANode nodes[dim];
	int cap;
	int size;
	int head;
	int tail;
	int firstEmpty;
public:
	Set()
	{
		this->cap = dim;
		this->size = 0;
		this->head = 0;
		this->tail = 0;
		this->firstEmpty = 0;
		for (int i = 0; i < dim; i++)
		{
			nodes[i] = DLLANode{ Artwork{} };
		}
	}
	//Set(const Set& s);
	~Set() {};

	DLLANode* getList();

	void addIfEmpty(Artwork a);
	void add(Artwork a);
	void searchFirstEmpty();
	void remove(Artwork a);
	bool find(Artwork a);
	int getSize();
	SetIterator getIterator();

	int getHead() { return head; }
	int getTail() { return tail; }
	
};


class SetIterator
{
private:
	DLLANode dlla[dim];
	int currentN;

public:
	SetIterator(Set s)
	{
		for (int i = 0; i < dim; i++)
			dlla[i] = s.getList()[i];
		currentN = s.getHead();
	}

	Artwork getCurrent() { return dlla[currentN].getInfo(); }
	bool valid();
	void next();

	int getCurrentIndex() { return currentN; }
};
